# useBool

`useBool` is a simple wrapper around `useState` for managing boolean values. It
provides a simple API for toggling the value, setting it to true, and setting it
to false.

```tsx
import { useBool, useToggle } from "@jobber/hooks/useBool";
```

<Canvas>
  <Story name="useBool">
    {() => {
      const [checked, on, off, toggle] = hooks.useBool(false);
      const [disabled, toggleDisabled] = hooks.useToggle(true);
      return (
        <Content>
          <Checkbox
            checked={checked}
            onChange={toggle}
            label={checked ? "On" : "Off"}
            disabled={disabled}
          />
          <br />
          <Button
            onClick={on}
            label={"On"}
            type={checked ? "primary" : "secondary"}
            disabled={disabled}
          />
          <br />
          <Button
            onClick={off}
            label={"Off"}
            type={checked ? "secondary" : "primary"}
            disabled={disabled}
          />
          <br />
          <Checkbox
            checked={disabled}
            onChange={toggleDisabled}
            label={"Disabled"}
          />
        </Content>
      );
    }}
  </Story>
</Canvas>
