# InputNumber

<ComponentStatus stage="pre" responsive="no" accessible="no" />

Input number is used in forms that accept numbers as an answer.

## Simple Example

<Playground>
  {() => {
    const [value, setValue] = useState(30);
    const handleChange = newValue => {
      setValue(newValue);
    };
    return (
      <InputNumber
        value={value}
        max={32}
        min={0}
        onChange={handleChange}
        placeholder="Age"
      />
    );
  }}
</Playground>

## Rendering the error message outside of the field

<Playground>
  {() => {
    const [value, setValue] = useState(2);
    const [errorMessage, setErrorMessage] = useState("");
    const maxLength = 4;
    const minLength = 0;
    const handleChange = newValue => {
      setValue(newValue);
      if (newValue < maxLength && newValue > minLength) {
        setErrorMessage("");
      }
    };
    const handleValidation = message => {
      setErrorMessage(message);
    };
    return (
      <>
        {errorMessage && <Text variation="error">{errorMessage}</Text>}
        <Text>
          Follow-up after
          <InputNumber
            name="durationInDays"
            value={value}
            size="small"
            inline={true}
            max={maxLength}
            min={minLength}
            onChange={handleChange}
            onValidate={handleValidation}
            align="center"
          />
          days
        </Text>
      </>
    );
  }}
</Playground>
