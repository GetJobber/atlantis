# Switch

## Parent controlled switch

<Playground>
  {() => {
    const [value, setValue] = useState(true);
    return (
      <>
        <button onClick={() => setValue(!value)}>Toggle</button>
        <br />
        <Switch value={value} onChange={newValue => setValue(newValue)} />
      </>
    );
  }}
</Playground>
