# Input Validation

<ComponentStatus stage="ready" responsive="yes" accessible="yes" />

`InputValidation` allows you to show the validation messages separate from a
particular input component.

```ts
import { InputValidation } from "@jobber/components/InputValidation";
```

<Playground>
  {() => {
    const [value, setValue] = useState("Juan");
    const [validationMessages, setValidationMessages] = useState(undefined);
    console.log("message", validationMessages);
    return (
      <>
        <InputValidation message={validationMessages} />
        <Text>
          My name is
          <InputText
            validations={{
              pattern: {
                value: /Jeff+/g,
                message: "Ship name must be Jeff"
              },
              minLength: {
                value: 5,
                message: "Ship name isn't long enough"
              },
              required: {
                value: true,
                message: "Ship must have a name"
              }
            }}
            onValidation={setValidationMessages}
            name="myName"
            value={value}
            size="small"
            inline={true}
            onChange={setValue}
            align="center"
            maxLength={4}
          />
        </Text>
      </>
    );
  }}
</Playground>
