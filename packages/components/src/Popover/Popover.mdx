  {() => {
    const checkboxRef = useRef();
    const [showPopover, setShowPopover] = useState(true);
    const [subscribed, setSubscribed] = useState(false);
    const onChangeSubscribed = () => {
      setShowPopover(false);
      setSubscribed(!subscribed);
    };
    return (
      <>
        <Checkbox
          defaultChecked={true}
          label="Subscribe on marketing emails"
          ref={checkboxRef}
          checked={subscribed}
          onChange={onChangeSubscribed}
        />
        <div>
          <Button
            label="Restart Scenario"
            onClick={() => setShowPopover(true)}
          />
        </div>
        <Popover
          attachTo={checkboxRef}
          open={showPopover}
          dismissible={true}
          placement={["right", "top"]}
          title="Show popover on load!"
          onRequestClose={() => setShowPopover(false)}
        >
          <Content>
            <Button
              label="Disappear forever"
              onClick={() => setShowPopover(false)}
            />
          </Content>
        </Popover>
      </>
    );
  }}
</Playground>

## Design & Usage Guidelines

When contributing to, or consuming the Popover component, consider the
following:

- When using the Popover component always consider the volume you would like it
  to be displayed at.
- The Popover component should never be used inside another component.
